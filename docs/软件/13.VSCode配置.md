## 配置终端使用Git ```BASH

```json
{
    "terminal.integrated.profiles.windows": {
        "Git Bash": {
            //"source": "Git Bash"
            "path": "d:\\software\\Git\\bin\\bash.exe"
        },
        "PowerShell": {
            "source": "PowerShell",
            "icon": "terminal-powershell"
        },
        "Command Prompt": {
            "path": [
                "${env:windir}\\Sysnative\\cmd.exe",
                "${env:windir}\\System32\\cmd.exe"
            ],
            "args": [],
            "icon": "terminal-cmd"
        },
    },
    "terminal.integrated.defaultProfile.windows": "Git Bash",
}
```
