使用Docker安装MySQL 

version: '3'
services:
  mysql:
    environment:
      MYSQL_ROOT_PASSWORD: ""
      MYSQL_USER: ''
      MYSQL_PASS: ''
    image: "docker.io/mysql:latest" 
    restart: always
    port:
      - 3306:3306
    volumes:
      - "./db:/var/lib/mysql"
      - "./conf/my.cnf:/etc/my.cnf"
      - "./init:/docker-entrypoint-initdb.d/"
